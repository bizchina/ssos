			if(){
				xxxx

			//good
			if () {
				xxxx
			}

			}

#####代码的换行使用

```php
	
	class User
	{	
		//demo
		public function getUser($id)
		{
			if ($id <= 0) return false;
			$dao = new Dao();
			$user = $dao -> getUser($id);
			return $user;
		}

		//better 不同逻辑间 适当换行 
		public function getUser($id)
		{
			if ($id <= 0) return false;

			$dao = new Dao();
			$user = $dao -> getUser($id);

			return $user;
		}
	}

```
